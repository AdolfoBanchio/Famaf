fun ExtremeKingfisherGame(playersNames : [1..N] of string ,lives :nat , k :nat)ret winner : String
    var q: queue of PLayer 
    p,aux : player
    rounds: nat 
    rounds := (lives * N) {-calculo la cantidad de rondas que se van a jugar segun cuantos jugadores hay y cuantas vidas tienen-}
    q:= empty_queue()
    for i:= 0 to N do
        p := createPlayer(playersNames[i],lives)
        enqueue(q,p)
    od {-lleno una cola con los jugadores, 
    los primeros del arreglo son los primeros de la cola, 
    por ende ultimos en irse-}

    for i:=1 to rounds do
        for j=1 to k do 
            aux := first(q) 
            dequeue(q)
            enqueue(q,aux)
        od 
        {-mando al fondo k veces las personas que esten al principio-}
        p := first(q) {-veo quien es el que quedo primero despues de la ronda y analizo sus vidas-}
        loseLife(p)
        if Â¬isAlive(p) then 
            dequeue(q) {-elimino el jugador que quedo al inicio de la cola-}
        fi
    od {-al terminar este bucle habra un solo jugador y sera el ganador-}
    aux := first(q)
    winner := getName(aux)
    destroy(q)
    destroy(p)
    destroy(aux)
endfun
